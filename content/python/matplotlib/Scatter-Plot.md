```python
# Importing Matplotlib library
from matplotlib import pyplot as plt
```

```python
# x-axis values 
x = [15, 20, 39, 44] 
```


```python
# y-axis values 
y = [10, 35, 58, 64] 
```


```python
# Function to plot x and y values
plt.scatter(x,y) 
# Function to show the plot 
plt.show()
```
